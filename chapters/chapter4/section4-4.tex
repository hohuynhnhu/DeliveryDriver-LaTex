\section{Đánh giá hệ thống}

\subsection{Ưu điểm}

\begin{itemize}
    \item Kiến trúc microservice giúp hệ thống dễ mở rộng
    \item Sử dụng Supabase giúp giảm độ phức tạp trong xác thực
    \item Thuật toán tối ưu lộ trình cải thiện hiệu quả giao hàng
\end{itemize}

\subsection{Hạn chế}

\begin{itemize}
    \item Thuật toán chưa tối ưu cho tập dữ liệu lớn
    \item Chưa triển khai hệ thống giám sát và logging nâng cao
\end{itemize}
